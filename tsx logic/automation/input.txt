							{shape === 'rect' && (
								<rect x="10" y="10" width={size - 20} height={size - 20} fill="none" stroke="#FFC107" strokeWidth="1" rx="8" />
							)}
							{shape === 'polygon' && (
								<polygon
									points={`${size / 2},10 ${size - 10},${size - 10} 10,${size - 10}`}
									fill="none"
									stroke="#9E9E9E"
									strokeWidth="1"
								/>
							)}

							opacity: pOp,
