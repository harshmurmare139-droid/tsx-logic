export const compositionConfig = {
  id: 'C2S1_AsliSamasya',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

  const { fps, durationInFrames } = useVideoConfig();

    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',

  const hourHandAngle = interpolate(frame, [0, durationInFrames], [0, 360], { extrapolateRight: 'clamp' });
  const minuteHandAngle = interpolate(frame, [0, durationInFrames], [0, 1440], { extrapolateRight: 'clamp' });

  // Red flood fill — danger rising from bottom
  const redFlood = interpolate(frame, [20, 80], [0, 0.18], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });

  // Title slam
  const titleScale = sp(10, { damping: 8, stiffness: 200, mass: 0.5 });
  const titleX = interpolate(sp(10), [0, 1], [-200, 0]);

  // Subtitle fade
  const subOp = interpolate(frame, [35, 55], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Screen shake on beat
  const shake = frame > 15 && frame < 20
    ? Math.sin(frame * 40) * 6
    : 0;

  // Particles
  const particles = Array.from({ length: 18 }, (_, i) => ({
    x: seededRandom(i * 7) * 1920,
    y: seededRandom(i * 13) * 1080,
    size: seededRandom(i * 3) * 6 + 2,
    speed: seededRandom(i * 17) * 0.4 + 0.1,
    op: seededRandom(i * 5) * 0.25 + 0.05,
  }));

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>

      {/* Gradient — deep ominous red-charcoal sweep */}
      <AbsoluteFill style={{
        background: 'radial-gradient(ellipse at 30% 50%, #2a0a0a 0%, #1C1C1C 55%, #0d0d0d 100%)',
      }} />

      {/* Red flood from bottom */}
      <AbsoluteFill style={{
        background: `linear-gradient(to top, rgba(255,0,0,${redFlood}) 0%, transparent 40%)`,
      }} />

      {/* Floating geometric shapes */}
      <AbsoluteFill style={{ transform: `translateX(${shake}px)` }}>
        {/* Large drifting circle — BG */}
        <svg style={{ position: 'absolute', top: -80, right: -80, opacity: 0.04 }} width="700" height="700">
          <circle cx="350" cy="350" r="340"
            fill="none" stroke="#ff0000" strokeWidth="2"
            transform={`rotate(${frame * 0.15}, 350, 350)`} />
          <circle cx="350" cy="350" r="260"
            fill="none" stroke="#e9d700" strokeWidth="1"
            transform={`rotate(${-frame * 0.2}, 350, 350)`} />
        </svg>

        {/* Triangle shape drifting */}
        <svg style={{
          position: 'absolute', bottom: 60, left: 40,
          opacity: 0.05,
          transform: `translateY(${Math.sin(frame * 0.03) * 12}px) rotate(${frame * 0.08}deg)`,
        }} width="300" height="300">
          <polygon points="150,10 290,280 10,280" fill="none" stroke="#FFC107" strokeWidth="2" />
        </svg>

        {/* Clock SVG — center-left */}
        <div style={{
          position: 'absolute',
          left: 160,
          top: '50%',
          transform: `translateY(-50%) scale(${sp(5, { damping: 14, stiffness: 100 })})`,
        }}>
          <svg width="320" height="320" viewBox="0 0 320 320">
            {/* Clock face */}
            <circle cx="160" cy="160" r="155" fill="#2D2D2D" stroke="#ff0000" strokeWidth="3" />
            <circle cx="160" cy="160" r="145" fill="none" stroke="#9E9E9E" strokeWidth="0.5" opacity="0.3" />

            {/* Hour markers */}
            {Array.from({ length: 12 }, (_, i) => {
              const angle = (i * 30 - 90) * (Math.PI / 180);
              const x1 = 160 + 125 * Math.cos(angle);
              const y1 = 160 + 125 * Math.sin(angle);
              const x2 = 160 + 140 * Math.cos(angle);
              const y2 = 160 + 140 * Math.sin(angle);
              return <line key={i} x1={x1} y1={y1} x2={x2} y2={y2} stroke="#F5F5F5" strokeWidth="3" />;
            })}

            {/* Hour hand */}
            <line
              x1="160" y1="160"
              x2={160 + 80 * Math.cos((hourHandAngle - 90) * Math.PI / 180)}
              y2={160 + 80 * Math.sin((hourHandAngle - 90) * Math.PI / 180)}
              stroke="#F5F5F5" strokeWidth="5" strokeLinecap="round"

              x1="160" y1="160"

            {/* Sand drain visual — bottom arc filling */}
            <path
              d={`M 30 290 A 140 140 0 0 0 290 290`}
              fill="none" stroke="#ff0000" strokeWidth="2" opacity="0.4"
              strokeDasharray="440"
              strokeDashoffset={440 - (frame / durationInFrames) * 220}

          position: 'absolute',
          right: 140,
          top: '50%',

          width: 900,
        }}>
          {/* Label */}
          <div style={{
            fontFamily: 'JetBrains Mono',
            fontSize: 22,
            color: '#ff0000',
            letterSpacing: 6,
            textTransform: 'uppercase',
            marginBottom: 24,
            opacity: subOp,
          }}>
            ◆ असली समस्या
          </div>

          {/* Hero text */}
          <div style={{
            fontFamily: 'Yantra One, Noto Sans Devanagari',
            fontSize: 210,
            fontWeight: 900,
            color: '#F5F5F5',
            lineHeight: 0.88,
            transform: `scale(${titleScale})`,
            transformOrigin: 'left center',
          }}>
            समय
          </div>

          <div style={{
            fontFamily: 'Yantra One, Noto Sans Devanagari',
            fontSize: 80,
            fontWeight: 700,
            color: '#e9d700',
            marginTop: 12,
            opacity: subOp,
            letterSpacing: -1,
          }}>
            अब तुम्हारे साथ नहीं
          </div>

          {/* Underline slash */}
          <div style={{
            marginTop: 28,
            height: 5,
            width: `${interpolate(frame, [40, 70], [0, 520], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' })}px`,
            background: 'linear-gradient(90deg, #ff0000, #e9d700)',

        </div>

        {/* Floating particles */}

            position: 'absolute',

            top: (p.y + frame * p.speed * 30) % 1080,
            width: p.size,
            height: p.size,
            borderRadius: '50%',

            opacity: p.op,

        {/* Stick figure — Buffett */}
        <div style={{
          position: 'absolute',
          bottom: 60,
          right: 160,
          transform: `translateY(${Math.sin(frame * 0.05) * 2}px) scale(${sp(45, { damping: 15, stiffness: 90 })})`,
