        position: 'absolute', top: 0, left: 0, right: 0, height: 12,

          top: `${seededRandom(i * 5) * 80}%`,

          opacity: 0.04 + seededRandom(i * 3) * 0.03,
          transform: `rotate(${frame * (i % 2 === 0 ? 0.1 : -0.08) + i * 30}deg)`,
        }} />
      ))}

      {/* Grid texture */}
      <AbsoluteFill style={{ opacity: 0.04 }}>
        <svg width="1920" height="1080">
          {[...Array(20)].map((_, i) => (
            <line key={`v${i}`} x1={i * 96} y1="0" x2={i * 96} y2="1080" stroke="#e9d700" strokeWidth="1" />
          ))}
          {[...Array(12)].map((_, i) => (
            <line key={`h${i}`} x1="0" y1={i * 90} x2="1920" y2={i * 90} stroke="#e9d700" strokeWidth="1" />
          ))}
        </svg>
      </AbsoluteFill>
